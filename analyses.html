# =========================
# 1. Preparar variable zona

study1_demo <- study1 %>%
  mutate(
    zona = case_when(
      tipo_zona == 1 ~ "Urban",
      tipo_zona == 2 ~ "Rural",
      TRUE ~ NA_character_
    )
  )%>%
  filter(!is.na(zona))

# =========================
# 2. Tamaño muestral
# =========================
n_zona <- study1_demo %>%
  count(zona, name = "N")

# =========================
# 3. Edad (media y sd)
# =========================
edad_zona <- study1_demo %>%
  group_by(zona) %>%
  summarise(
    mean_sd_age = paste0(mean(age, na.rm = TRUE)%>%round(0)," (", sd(age, na.rm = TRUE)%>%round(0), ")"),
    .groups = "drop"
  )

# =========================
# 4. Nivel educacional
# =========================
educ_zona <- study1_demo %>%
  count(zona, educ_level) %>%
  group_by(zona) %>%
  mutate(pct = round(100 * n / sum(n), 1)) %>%
  ungroup()

# =========================
# 5. Algorithmic literacy
# =========================
alg_lit_zona <- study1_demo %>%
  group_by(zona) %>%
  summarise(
    alg_lit = paste0(sum(alg_lit == 1, na.rm = TRUE)," (",round(100 * mean(alg_lit, na.rm = TRUE), 1), ")"),
    .groups = "drop"
  )

# =========================
# 6. Diferencia de medias
# =========================
alg_aware_diff <- with(
  study1_demo,
  t.test(alg_aware ~ zona)
)

diff_table <- tibble(
  diff_means = alg_aware_diff$estimate[1] - alg_aware_diff$estimate[2],
  p_value    = alg_aware_diff$p.value
)

alg_aware_table <- data.frame(
  zona = c("Rural", "Urban"),
  alg_aware_mean = c("2.2 (p > 0.01)", "2.8")
)

# =========================
# 7. Gender by zone
# =========================
gender_zona <- study1_demo %>%
  mutate(
    gender_lab = case_when(
      gender == 1 ~ "Women",
      gender == 0 ~ "Men",
      TRUE ~ NA_character_
    )
  ) %>%
  count(zona, gender_lab) %>%
  group_by(zona) %>%
  mutate(
    gender_pct = round(100 * n / sum(n), 1),
    gender_freq_pct = paste0(n, " (", gender_pct, "%)")
  ) %>%
  ungroup() %>%
  select(zona, gender_lab, gender_freq_pct) %>%
  pivot_wider(
    names_from = gender_lab,
    values_from = gender_freq_pct
  )

gender_tabla <- data.frame(
  zona = c("Rural", "Urban"),
  gender = c("M: 184 (41%) / W: 265 (59%)", "M: 327 (43.5%) / W: 424 (56.5%)")
)

# =========================
# 8. Educational level by zone
# =========================

  mutate(
    educ_lab = case_when(
      educ_level == 1 ~ "Basic education",
      educ_level == 2 ~ "Middle school (incomplete)",
      educ_level == 3 ~ "Middle school (complete)",
      educ_level == 4 ~ "Technical higher education",
      educ_level == 5 ~ "University / Postgraduate",
      TRUE ~ NA_character_

  ) %>%
  filter(!is.na(educ_lab)) %>%
  count(zona, educ_lab) %>%
  group_by(zona) %>%
  mutate(
    educ_pct = round(100 * n / sum(n), 1),
    educ_item = paste0(educ_lab, ": ", n, " (", educ_pct, "%)")
  ) %>%
  summarise(
    Education = paste(educ_item, collapse = "; "),
